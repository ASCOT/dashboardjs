158 /**
159  * Parses a string that represents a coordinate to return a CelestialCoordinate
160  * with the specified RA and DEC. The parser is supposed to be fairly tolerant, as it
161  * will be translating from a web text input. The input can be in the form of HMS or decimal degrees.
162  * @param {String} Expects a coordinate repesented by RA and DEC in H M S or decimal degrees as <br/>
163  *    00 00 00.0 +00 00 00.0  <br/>
164  * or 00 00 00.0 -00 00 00.0  <br/>
165  * or 00:00:00.0 +00:00:00.0  <br/>
166  * or two decimal numbers     <br/>
167  * where 0 is any digit.      <br/>
168  * Decimals not required, but a + or - is required for the DEC.
169  * @return {CelestialCoordinate#} A CelestialCoordinate if text string represents an RA and DEC
170  * otherwise returns null.
171 */
172 UW.astro.CelestialCoordinate.parseCoord = function(text) {
173     // try to split the string into ra and dec at +/-
174     var isDecPositive = (text.indexOf('+') != -1);
175     var splitRegex = /\+|\-/;
176     var raDec = text.split(splitRegex);
177     if (!(raDec.length === 2)) {
178     	return null;
179     }
180     
181     // split ra and dec at spaces
182     var ra = tupleStrToDecimal(raDec[0]);
183     var dec = tupleStrToDecimal(raDec[1]);
184     if (ra === null || dec === null) {
185 	    return null;
186     }
187     if (!isDecPositive) {
188 	    dec = dec * -1;
189     }
190     if (!((ra >= 0) && (ra <= 360))) {
191 	    return null;
192     }
193     if (!(dec >= -90) && (dec <= 90)) {
194     	return null;
195     }
196     // seems like an ok coordinate, so set this CelestialCoord as these coords are return it
197     return new UW.astro.CelestialCoordinate(ra, dec);
198 }

200 /**
201  * @private 
202  * @returns A decimal representation of the coordinate string
203  * @param {String} A string representing 2 ints and a float, or just a single float
204  */
205 function tupleStrToDecimal(str) {
206     str = trim(str);
207     var singleFoat = false;
208     // split ra and dec at spaces
209     var tupleStr = str.split(/ /); 
210     if (!(tupleStr.length === 3)) {
211 	    tupleStr = str.split(/:/); // split ra and dec at colons
212 	    if (!(tupleStr.length === 3)) {
213 	        var tryFloat = parseFloat(tupleStr[0]); // maybe it's only a single decimal?
214 	        if (tupleStr.length === 1 && !isNaN(tryFloat)) {
215     		    return tryFloat;
216 	        } else {
217 	    	    return null;
218 	        }
219 	    }
220     }
221     // test for whether the first 2 are ints, and the 3rd is a float
222     var tupleNums = [];
223     // specify radix of 10, beacuse 05  looks like octal 5
224     tupleNums[0] = parseInt(tupleStr[0], 10);
225     tupleNums[1] = parseInt(tupleStr[1], 10);
226     tupleNums[2] = parseFloat(tupleStr[2], 10);
227     // test for NaN in all 3 fields of both ra and dec
228     for (var i=0; i<3; i++) {
229     	if (isNaN(tupleNums[i])) {
230 	        return null;
231 	    }
232     }
233     // convert the 3 sections to decimal
234     return tupleNums[0] + (tupleNums[1] / 60.0) + (tupleNums[2] / 3600.0);
235 }

302 // Accepts an object, and attempts to parse it to 
303 // return either null or a CelestialCoordinate
304 // TODO: add more parsing
305 UW.astro.CelestialCoordinate.prototype.parseObject = function(obj) {
306 	if (obj.hasOwnProperty('ra') && obj.hasOwnProperty('dec')) {
307 		cc = new UW.astro.CelestialCoordinate(obj.ra, obj.dec);
308 		var desc = [];
309 		for (var property in obj) {
310 		    if (obj[property] !== null && obj[property] !== undefined) {
311 			desc.push(property + ": " + obj[property].toString());
312 		    }
313 		}
314 		cc.setDescription(desc.join('\n'));
315 		if (obj.hasOwnProperty('db_visible')) {
316 		    cc.setVisible(obj.db_visible);
317 		}
318 		if (obj.hasOwnProperty('db_color')) {
319 		    cc.setColor(obj.db_color);
320 		}
321 		return cc;
322 	} else {
323 	    return null;
324 	}
325 }
