 39 /**
 40 * Create a DataInterface object to allow a gadget to communicate with the DataGadget
 41 * This object provides an interface to allow a gadget to communicate with the DataGadget.
 42 * When instantiated, the interface searches on the parent for the iframe of the DataGadget 
 43 * and saves the actual DataGadget object so that methods can be called on it.
 44 * Because we can't assume the DataGadget is the first iframe loaded, the search occurs asyncronously
 45 * with a DELAY and MAX_RETRIES set as constants below.
 46 *
 47 * This interface provides a registerMe() function which finds the DataGadget, registers the gadget
 48 * to a given namespace, registers all variables in the variable list, and registers all triggers in
 49 * the trigger list.
 50 *
 51 * Example code to register a gadget 
 52 * @class DataInterface
 53 * @constructor
 54 * @param {string} gadget Name of gadget creating this DataInterface
 55 * @returns {object} Returns a DataInterface object
 56 */ 
 57 UW.astro.DataInterface = function(gadget){
 58     var my = new Object();
 59     
 60     // Constants
 61     var DATAGADGET_TITLE = "DataGadgetv0.1",  // This is the name that will be searched 
 62                                               // for as the iframe of the DataGadget, so it must
 63                                               // match the iframe name of the datagadget or it will
 64                                               // never be found.
 65         DATAGADGET_OBJECT_NAME = "dg",
 66         FIND_GADGET_DELAY = 1000, // number of seconds to delay before 
 67                                   // retrying to find the datagadget
 68         FIND_GADGET_MAX_RETRIES = 20, // max retries to find DG before accepting failure
 69         MM_TIMER_DURATION = 4; // duration to display notifications in seconds
 70 
 71     // Private
 72     var _gadgetName = gadget,
 73         _namespaceList = [],    // namespaces to associate with
 74         _dataGadget = null,
 75         findGadgetRetries = 0;  // incremented each time we retry the search for the DataGadget
 76    
 77     var miniMsg = null; 
 78     // make the message for a MiniMessage writable by the whole class
 79     var msg = null;
 80     
 81     // EXPOSED REGISTRATION
 82     /**
 83      * @name registerMe
 84      * @memberOf UW.astro.DataInterface#
 85      * @type void
 86      * @function
 87      * @description Register a gadget to a namespace. Optionally register variables and triggers at the same time.
 88      * This function runs asyncronously, and finishes by executing callback functions (one for
 89      * success or one for failure).                                       <br/>
 90      * NOTE: Variables are passed as object properties.                   <br/>
 91      *        ex: registerMe( { var1 : value1, var2 : value2 } );
 92      * 
 93      * @param {function} successCallback *Optional* Function (in gadget context) executed 
 94      *        on successful registration.
 95      * @param {function} failureCallback *Optional* Function (in gadget context) executed 
 96      *        on failed registration.
 97      * @param {string[]} namespaceList *Required* Array of namespace names in which to register 
 98      *        this gadget and variables and triggers.
 99      * @param {string[]} variableList *Optional* Array of variable names to register the gadget to
100      * @param {array.<array.<string, function>>} triggerList *Optional* Array of trigger arrays, where 
101      *        each trigger array is an array containing the variable name getting trigger, and the 
102      *        trigger function for that variable.                           <br/>
103      *        Each trigger is a 2-element array as follows:                 <br/>
104      *                [ "VariableName", triggerFunction ]                   <br/> 
105      *        example with one trigger: (notice the array inside the array) <br/>
106      *                var triggerArray = [ ["Viewport_Center_Coordinate", function(){ zoomMe() } ] ] 
107      * @param {MiniMessage} miniMsg *Optional* Allows a gadget to pass an existing MiniMessage
108      *        object if one has already been created. Otherwise one is created to display registration
109      *        progress.
110      */
111     my.registerMe = function(oArg) {
112         // transfer variables from argument object
113         var successfulCallback = 'successCallback' in oArg ? oArg.successCallback : function(){};
114         var failureCallback = 'failureCallback' in oArg ? oArg.failureCallback : function(){};
115         var namespaceList = 'namespaceList' in oArg ? oArg.namespaceList : [];
116         var variableList = 'variableList' in oArg ? oArg.variableList : [];
117         var triggerList = 'triggerList' in oArg ? oArg.triggerList : [];
118         // miniMsg already globally defined
119         miniMsg = 'miniMsg' in oArg? oArg.miniMsg : new gadgets.MiniMessage();
120        
121         // create a MiniMessage static message to notify about registration
122         msg = miniMsg.createStaticMessage("Attempting to register gadget...");
123         
124         // TODO: more variable validation
125         if (!(namespaceList.length >= 1)) {
126             debugLog('DI: No namespace provided');
127             failureCallback();
128             return;
129         }
130         if (!_gadgetName.length || _gadgetName.length < 1){
131             debugLog('DI: No gadget name provided');
132             failureCallback();
133             return;
134         }
135         // TODO: check for a valid miniMsg
136    
137         // set the (global) _namespaceList
138         _namespaceList = namespaceList;
139         
140         // register the gadget with each namespace in the list
141         // for each namespace in list, it starts a new asynchronous registerer
142         for (var i=0, len=namespaceList.length; i<len; i++){
143             var nsName = namespaceList[i];
144             registerGadget(nsName,
145                             // successful callback
146                             function(){
147                                 // continue the registration for this particular namespace
148                                 continueRegistration(nsName, variableList, triggerList, successfulCallback)
149                             },
150                             // failure callback
151                             function(){
152                                 registrationFailed(failureCallback) // call the local failure callback, and pass it the gadgets failure callback
153                             }
154                             );
155         }
156     }
157    
158     /** @private */
159     // Registers the variables and the triggers for a namespace
160     // then calls the successful callback.
161     // This function doesnt get a failure callback, so never calls one
162     function continueRegistration(nsName, variableList, triggerList, successCallback){
163         debugLog('DI: Continuing registration in namespace: ' + nsName);
164         // register variables
165         for (var i=0, len=variableList.length; i<len; i++){
166             if (registerVarInNamespace(nsName, variableList[i], null)){
167                 debugLog('DI: Successfully registered variable "' + variableList[i] + '"');
168             }
169             else {
170                 debugLog('DI: FAILED to register variable "' + variableList[i] + '"');
171             }
172         }
173         
174         // register triggers
175         for (var i=0, len=triggerList.length; i<len; i++){
176             // triggerList[i][0] is trigger name
177             // triggerList[i][1] is the function
178             if (my.registerTrigger(nsName, triggerList[i][0], triggerList[i][1]) ){
179                 debugLog('Successfully registered trigger for "' + triggerList[i][0] + '."');
180             }
181             else {
182                 debugLog('FAILED to register trigger for "' + triggerList[i][0] + '."');
183             }
184         }
185         successCallback();
186     }
187 
188     /** @private */
189     // print a failure message, and then call the gadgets failure callback
190     function registrationFailed(failureCallback) {
191         debugLog('DI: Registration Failed.');
192         miniMsg.dismissMessage(msg);
193         miniMsg.createDismissibleMessage("Unable to register gadget.\nWe cannot communicate with other gadgets.");
194         gadgets.window.adjustHeight();
195         // call additional failure callback supplied by gadget
196         failureCallback();
197     }

199     /** @private */
200     // Attempts to find the iframe of DataGadget using browser-specific methods,
201     // and then return its actual datagadget object (so datagadget methods can be called)
202     // Returns reference to DataGadget object, or null if not found
203     function lookForDataGadget(){
204         debugLog('DI: Looking for dg...');
205         var parentFrames = parent.document.getElementsByTagName("iframe");
206         for (i=0; i<parentFrames.length; i++){
207             // get the frame title to check if it's the DataGadget
208             var f = parentFrames[i],
209                 title = "";
210             if (f.contentDocument) {            // DOM (not IE)
211                 title = f.contentDocument.title;
212             }
213             else if (f.contentWindow) {
214                 title = f.contentWindow.title;  // IE only
215             }
216             if (title === DATAGADGET_TITLE){
217                 var fName = parentFrames[i].name;
218                 // DATAGADGET_OBJECT_NAME is the actual DataGadget object name in the DataGadget iframe
219                 // (the object is a property of the iframe object because it is defined at the root
220                 // level of the html file)
221                 var dg = parent.window.frames[fName][DATAGADGET_OBJECT_NAME];
222                 if ((dg !== null) && (dg !== undefined)) {
223                     debugLog('DI: DataGadget found.');
224                     return dg;             
225                 }
226             }
227         }
228         debugLog('DI: DataGadget not found yet...');
229         return null;
230     }

232     /** @private */    
233     // Begins asyncronous search for datagadget. Keeps retrying FIND_GADGET_MAX_RETRIES number of times,
234     // separated by a delay of FIND_GADGET_DELAY seconds.
235     // Returns true if found, false if not
236     function findDataGadget(successCallback, failCallback){
237         debugLog('DI: findDataGadget begins.');
238         _dataGadget = lookForDataGadget();
239        
240         if (_dataGadget !== null) {
241             // Success! We have set the global dataGadget variable to the dg object.
242             debugLog('DI: findDataGadget: Success! firing successCallback...');
243             successCallback();
244             return true;
245         }
246         else {
247             // call this same function again until we find the datagadget
248             
249             debugLog('DI: findDataGadget: dg is null. Starting timer to find it...');
250             if (findGadgetRetries < FIND_GADGET_MAX_RETRIES) {
251                 findGadgetRetries++;    // increment to keep track of our attempts
252                 debugLog('DI: findDataGadget: beginning retry ' + findGadgetRetries);
253                 window.setTimeout(function(){ debugLog('DI: findDataGadget timeout fired!'); findDataGadget(successCallback, failCallback);}, FIND_GADGET_DELAY);
254             }
255             else { // We have exceed our retry limit. Just give up and call the failCallback
256                 debugLog('DI: findDataGadget: I accept failure. Calling failCallback...');
257                 failCallback();
258             }
259             
260 //          alert("Cannot find " + DATAGADGET_TITLE + "\nPlease add the gadget or there will be problems with gadget-to-gadget communication");
261             return false;
262         }
263     }

265     /** @private */
266     // Returns true if the global _dataGadget variable is not null, otherwise, it attempts to find
267     // it and returns false if it cant.
268     function dgExists(successCallback, failCallback){
269         if (_dataGadget !== null){
270             debugLog('DI: dgExists: dg EXISTS.');
271             return true;
272         }
273         else {
274             debugLog('DI: dgExists: dg = null. Running findDataGadget()...');
275             // we set the retries (global) to 0
276             // not a big deal if this gets reset while we are already retrying
277             findGadgetRetries = 0;
278             return findDataGadget(successCallback, failCallback); // returns true if found, false if not
279         }
280     }

282     /** @private */
283     // Begins the gadget registration. At this point, the DataGadget may or may not have been found, 
284     // so calling dgExists will find it, or not and call the failCallback
285     function registerGadget(namespaceName, successCallback, failCallback){
286         debugLog('DI: registerGadget: begins.');
287         // dgExists takes 2 callbacks as args, a success, and a fail
288         // for the success, we send doRegisterGadget (which registers and calls successCallback)
289         // for the fail, we send failedRegisterGadget (which just calls failCallback)
290         return dgExists(
291                 function(){doRegisterGadget(namespaceName, successCallback);},
292                 function(){failedRegisterGadget(namespaceName, failCallback);}
293                 ); 
294     }
295     
296     /** @private */
297     // Last function called to register gadget. This is the function that calls the 
298     // DataGadget's registerGadget() function, then notifies by MiniMessage and calls the
299     // successCallback.
300     function doRegisterGadget(namespaceName, successCallback){
301         debugLog('DI: doRegGadget: dg does exist, so actually registering "' + _gadgetName + '" in "' + namespaceName + '" now...');
302         _dataGadget.registerGadget(_gadgetName, namespaceName);
303         
304         // notify by MiniMessage
305         if (msg !== null) {
306             miniMsg.dismissMessage(msg);
307             // free up msg
308             msg = null;
309             miniMsg.createTimerMessage("Registration Succesful.", MM_TIMER_DURATION);
310             gadgets.window.adjustHeight();
311         }
312         
313         successCallback();
314     }

325     // Internal function to register a single variable in a single namespace
326     function registerVarInNamespace(nsName, varName, varValue) {
327         if (!_dataGadget.registerVariable(_gadgetName, nsName, varName, varValue)) {
328             debugLog('DI: my.registerVariable: There was a problem registering gadget "' + _gadgetName + '" to variable "' + varName + '" in "' + _namespaceList[i]);
329             return false;
330         } else {
331             return true;
332         }

335     // PUBLIC METHODS
336     /**
337      * @name registerVariable
338      * @memberOf UW.astro.DataInterface#
339      * @function
340      * @description Tries to register this gadget to a variable in each namespace that
341      * the gadget is currently registered to. 
342      * NOTE: A gadget is not able to read or write to a datagadget variable if it is not registered 
343      * to the variable (and registered to the namespace of the variable).
344      * @param {string} varName Name of variable to register.
345      * @param {object} varValue Initial value of variable to set. If the variable already exists,
346      *      then the value is NOT set. The gadget is simply registered to the variable.
347      * @return {boolean} Returns true if variable was successfully registered in all applicable
348      *      namespaces, false if anything went wrong. If something went wrong, check the log.
349      */
350     my.registerVariable = function(varName, varValue){
351         // TODO: function to check if dg !== null
352         var success = true;
353         if (_dataGadget !== null) {
354             for (var i=0, len=_namespaceList.length; i<len; i++) {
355                 if (!registerVarInNamespace(_namespaceList[i], varName, varValue)) {
356                     success = false;
357                 }
358             }
359         } else {
360             return false;
361         }
362         return success;
363     }

365    /** @name writeVariable
366      * @memberOf UW.astro.DataInterface#
367      * @function
368      * @description Writes a new value to a datagadget variable.
369      * NOTE: Writing to a variable fires all triggers registered to that variable EXCEPT for
370      * triggers registered by the gadget that writes the variable.
371      * @param {string} namespaceName The namespace of the variable to write.
372      * @param {string} varName Name of the variable to write.
373      * @param {object} varValue Value to set the variable to.
374      * @return {boolean} Returns true if successful, false if not successful.
375     */
376    my.writeVariable = function(namespaceName, varName, varValue){
377         if (_dataGadget !== null) {
378                 if (_dataGadget.writeVariable(_gadgetName, namespaceName, varName, varValue)) {
379                     return true;
380                 }
381                 else {
382                     debugLog('DI: writeVariable: Problem writing variable "' + varName + '"!');
383                     return false;
384                 }
385         }

387     }

409      * @memberOf UW.astro.DataInterface#
410      * @function
411      * @description Registers a trigger to a specific datagadget variable.
412      * @param {string} namespaceName The namespace of the variable to read.
413      * @param {string} varName The name of the variable on which to register a trigger.
414      * @param {function} triggerCallback The function to be executed when the trigger is fired. NOTE: The

417      * @return {boolean} Returns true if successful, false if not successful.

419     my.registerTrigger = function(namespaceName, varName, triggerCallback){
420         if (_dataGadget !== null) {
421             if (_dataGadget.registerTrigger(_gadgetName, namespaceName, varName, triggerCallback)) {
422                 return true;
423             }
424             else {

426                 return false;
427             }
428         }
