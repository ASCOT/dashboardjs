 17 /**
 18  * Create a new Dataset.
 19  * @class Represents a table of values, with rows and columns. 
 20  * Each row has unique id, a color, and a visibility, 
 21  * @constructor
 22  * @param {object[]} json A JSON array from which the dataset is created.
 23  *      The JSON should be of the form of an array of objects where each property
 24  *      is a column name, and each value is the value of that row and column.
 25  *      First object in array is first row in dataset.
 26  */
 27 UW.astro.Dataset = function(json) {
 28 	this.dsName = "";	
 29 	this.visible = true;
 30 	
 31 	// TODO: verify json is an array of objects (in json form)
 32 	this.db = new TAFFY(json);
 33 	
 34 	// add a color column to the db
 35 	if (!this.hasColor()) {
 36 		this.db.update({db_color:""}); // no where clause, so updates all records
 37 	}
 38 
 39 	// add db index to each record (overwrite if it already exists
 40 	//if (!this.hasColumn("db_id")) {
 41 		this.db.update({db_id:""});
 42 		this.db.forEach( function(rec,index){ rec.db_id = index; return rec; } );
 43 	//}
 44 	
 45 	// add db visibility to each record
 46 	if (!this.hasColumn("db_visible")) {
 47 		// set all records visble=true
 48 		this.db.update({db_visible:true});
 49 	}
 50 
 51 	// add CelestialCoord if possible
 52 	if (this.hasColumn('ra') && this.hasColumn('dec')) {
 53 		this.db.forEach(function(rec,index){
 54 							cc = new UW.astro.CelestialCoordinate;
 55 							// set the cc id to be "<datasetName>_<index>"
 56 							var cc =  UW.astro.CelestialCoordinate.prototype.parseObject(rec);
 57 							cc.setId("dsName_" + index);
 58 							rec.db_CelestialCoord = cc; 
 59 							return rec;
 60 						} );
 61 	}
 62 };

 94 UW.astro.Dataset.prototype.getNonDBColumns = function() {
 95 	var columns = this.getColumns();
 96 	for (var i=columns.length -1; i>=0; i--) {
 97 		if (columns[i].indexOf('db_') === 0) {
 98 			columns.splice(i,1);
 99 		}
100 	}
101 	return columns;
102 }

108 UW.astro.Dataset.prototype.getColoredIndexArray = function() {
109     // make a list of ever index
110     var list = [];
111 	var entireDb = this.db.get();
112 	for (var i=0, len=entireDb.length; i<len; i++) {
113 	    list[i] = [ entireDb[i]["db_id"], entireDb[i]["db_color"] ];
114 	}

118 // returns an array of the unique color values in the dataset
119 UW.astro.Dataset.prototype.getUniqueColors = function() {
120 	var colorList = [];
121 	var entireDb = this.db.get();
122 	for (var i=0, len=entireDb.length; i<len; i++) {
123 		if (!arrayContains(colorList, entireDb[i]["db_color"])) {
124 			colorList.push(entireDb[i]["db_color"]);
125 		}
126 	}
127 	return colorList.sort();
128 }

168 // Accepts an array of indexes and string that defines color
169 // Sets the color of all indexes to specified color
170 UW.astro.Dataset.prototype.setIndexesColor = function(indexes, color) {
171 	// TODO: how to change db_color to this.DB_COLOR (const)?
172 	this.db.update({db_color:color}, indexes);
173 	// set the color change for all these indexes in the celestial coord
174 	this.db.forEach(
175 					function(rec,index){ 
176 						if (rec.hasOwnProperty('db_CelestialCoord')) {
177 							var cc = rec.db_CelestialCoord;
178 							if (cc !== null) {
179 								cc.setColor(color);
180 								rec.db_CelestialCoord = cc;
181 								return rec;
182 							}
183 						}
184 					}, {db_id: indexes});
185 }

224 //HELPER
225 function arrayContains(arr, el) {
226 	for (var i=0, len=arr.length; i<len; i++) {
227     	if (arr[i] === el) {
228         	return true;
229     	}
230     }
231     return false;
232 }
